# Project 1: CRUD Todo List (Beginner)

> Aplikasi manajemen tugas yang berinteraksi langsung dengan smart contract

---

## 📋 Overview

**Tingkat:** Beginner  
**Tujuan:** Memahami read/write operations dan event listening di wagmi

### Apa yang Akan Dipelajari

- Menulis data ke smart contract (write operations)
- Membaca data dari smart contract (read operations)
- Mendengarkan event untuk real-time updates
- Integrasi Next.js dengan wagmi/viem
- Handling loading states dan error states

---

## 🎯 Fitur Utama

| Fitur | Tipe | Deskripsi |
|-------|------|-----------|
| Tambah tugas | Write | Membuat task baru di smart contract |
| Lihat daftar tugas | Read | Mengambil semua task dari blockchain |
| Update status | Write | Toggle status selesai/tidak selesai |
| Real-time update | Event | Update UI saat ada perubahan data |

```solidity
contract TodoList {
```

### Implementasi Lengkap

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

contract TodoList {
    struct Task {
        uint256 id;
        string content;
        bool completed;
        address owner;
        uint256 createdAt;
    }

    event TaskCreated(uint256 indexed id, string content, address indexed owner);
    event TaskToggled(uint256 indexed id, bool completed, address indexed owner);

    mapping(uint256 => Task) public tasks;
    uint256 public nextTaskId;
    mapping(address => uint256[]) public userTaskIds;

    function createTask(string calldata _content) external {
        require(bytes(_content).length > 0, "Content cannot be empty");
        
        tasks[nextTaskId] = Task({
            id: nextTaskId,
            content: _content,
            completed: false,
            owner: msg.sender,
            createdAt: block.timestamp
        });
        
        userTaskIds[msg.sender].push(nextTaskId);
        
        emit TaskCreated(nextTaskId, _content, msg.sender);
        nextTaskId++;
    }

    function toggleTask(uint256 _taskId) external {
        Task storage task = tasks[_taskId];
        require(task.owner == msg.sender, "Not the task owner");
        
        task.completed = !task.completed;
        
        emit TaskToggled(_taskId, task.completed, msg.sender);
    }

    function getUserTasks(address _user) external view returns(Task[] memory) {
        uint256[] storage ids = userTaskIds[_user];
        Task[] memory result = new Task[](ids.length);
        
        for (uint256 i = 0; i < ids.length; i++) {
            result[i] = tasks[ids[i]];
        }
        
        return result;
    }
}
```

```solidity
// test/TodoList.t.sol
pragma solidity ^0.8.24;

import {Test, console} from "forge-std/Test.sol";
import {TodoList} from "../src/TodoList.sol";

contract TodoListTest is Test {
    TodoList public todoList;
    address public user1 = address(0x1);
    address public user2 = address(0x2);

        todoList = new TodoList();
    }

    function test_createTask() public {
        vm.prank(user1);
        todoList.createTask("Learn Solidity");
        
        (uint256 id, string memory content, bool completed, address owner, uint256 createdAt) = 
            todoList.tasks(0);
        
        assertEq(id, 0);
        assertEq(content, "Learn Solidity");
        assertEq(completed, false);
        assertEq(owner, user1);
    }

    function test_toggleTask() public {
        vm.prank(user1);
        todoList.createTask("Learn Solidity");
        
        vm.prank(user1);
        todoList.toggleTask(0);
        
        (, , bool completed, , ) = todoList.tasks(0);

        vm.prank(user1);
        todoList.createTask("Task 2");
        
        vm.prank(user2);
```

```typescript
'use client'

import { useAccount, useConnect, useDisconnect } from 'wagmi'
import { useState } from 'react'

export function CustomConnectButton() {
  const { address, isConnected } = useAccount()

  const { disconnect } = useDisconnect()

  const formatAddress = (addr: string) => {

          Disconnect

    <div>
      <button
        onClick={() => setShowModal(true)}
        className="bg-purple-600 text-white px-4 py-2 rounded-lg hover:bg-purple-700"
```

```typescript
import { useReadContract } from 'wagmi'
import { useAccount } from 'wagmi'
import { todoListAbi } from '@/abi/todo'

const CONTRACT_ADDRESS = '0x...' // Deploy contract dan isi alamatnya

export function useTasks() {
  const { address } = useAccount()

    abi: todoListAbi,
```

#### Create Task

```typescript
// hooks/useCreateTask.ts
import { useWriteContract, useWaitForTransactionReceipt } from 'wagmi'
import { todoListAbi } from '@/abi/todo'

const CONTRACT_ADDRESS = '0x...'

export function useCreateTask() {
  const { data: hash, writeContract, isPending } = useWriteContract()
  
  const { isLoading: isConfirming, isSuccess } = 
    useWaitForTransactionReceipt({ hash })
  
  function createTask(content: string) {
    writeContract({
      address: CONTRACT_ADDRESS,
      abi: todoListAbi,
      functionName: 'createTask',
      args: [content],
    })
  }
  
  return {
    createTask,
    hash,
    isPending,
    isConfirming,
    isSuccess,
  }
}
```

### AddTodoForm Component

```typescript
// components/AddTodoForm.tsx
'use client'

import { useState } from 'react'
import { useCreateTask } from '@/hooks/useCreateTask'

export function AddTodoForm({ onSuccess }: { onSuccess: () => void }) {
  const [content, setContent] = useState('')
  const { createTask, isPending, isConfirming, isSuccess } = useCreateTask()
  
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    if (!content.trim()) return
    
    createTask(content)
    
    if (isSuccess) {
      setContent('')
      onSuccess()
    }
  }
  
  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        value={content}
        onChange={(e) => setContent(e.target.value)}
        placeholder="What needs to be done?"
        disabled={isPending || isConfirming}
      />
      <button type="submit" disabled={isPending || isConfirming}>
        {isPending ? 'Sign...' : isConfirming ? 'Confirming...' : 'Add Task'}
      </button>
    </form>
  )
}
```

### TodoItem Component

```typescript
// components/TodoItem.tsx
'use client'

import { useToggleTask } from '@/hooks/useToggleTask'

interface Task {
  id: bigint
  content: string
  completed: boolean
  owner: string
  createdAt: bigint
}

export function TodoItem({ task, onToggle }: { task: Task; onToggle: () => void }) {
  const { toggleTask, isConfirming } = useToggleTask()
  
  const handleToggle = () => {
    toggleTask(task.id)
    onToggle()
  }
  
  return (
    <div className={task.completed ? 'completed' : ''}>
      <input
        type="checkbox"
        checked={task.completed}
        onChange={handleToggle}
        disabled={isConfirming}
      />
      <span>{task.content}</span>
    </div>
  )
}
```

```typescript
// components/TodoList.tsx
'use client'

import { useAccount, useConnect } from 'wagmi'
import { useTasks } from '@/hooks/useTasks'
import { TodoItem } from './TodoItem'

import { ConnectButton } from '@rainbow-me/rainbowkit'
```
