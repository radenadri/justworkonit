```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

import {ERC20} from "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import {ERC20Permit} from "@openzeppelin/contracts/token/ERC20/extensions/ERC20Permit.sol";
import {ERC20Votes} from "@openzeppelin/contracts/token/ERC20/extensions/ERC20Votes.sol";

contract GovernanceToken is ERC20, ERC20Permit, ERC20Votes {
    
    uint256 public constant INITIAL_SUPPLY = 1_000_000 * 10 ** decimals();
    
    constructor() ERC20("GovernanceToken", "GOV") ERC20Permit("GovernanceToken") {
        _mint(msg.sender, INITIAL_SUPPLY);
    }
    
    // Override required by Solidity
    function mint(address to, uint256 amount) external {
        _mint(to, amount);
    }
    
    function burn(address from, uint256 amount) external {
        _burn(from, amount);
    }
    
    // Overrides required by ERC20Votes
    function _update(
        address from,
        address to,
        uint256 value
    ) internal override(ERC20, ERC20Votes) {
        super._update(from, to, value);
    }
    
    function nonces(
        address owner
    ) public view override(ERC20Permit, Nonces) returns (uint256) {
        return super.nonces(owner);
    }
}
```

### GovernorContract.sol

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

import {Governor} from "@openzeppelin/contracts/governance/Governor.sol";
import {ERC20} from "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import {IVotes} from "@openzeppelin/contracts/governance/utils/IVotes.sol";

contract DAOGovernor is Governor {
    ERC20 public immutable token;
    
    // Governance parameters
    uint256 public votingDelay = 1 days;
    uint256 public votingPeriod = 7 days;
    uint256 public proposalThreshold = 1 * 10 ** 18;
    uint256 public quorum = 4; // 4% of tokens
    
    constructor(ERC20 _token) {
        token = _token;
    }
    
    // votingDelay: How long after proposal is created can people vote
    function votingDelay() public view override returns (uint256) {
        return votingDelay;
    }
    
    // votingPeriod: How long does the voting last
    function votingPeriod() public view override returns (uint256) {
        return votingPeriod;
    }
    
    // proposalThreshold: How many tokens needed to create proposal
    function proposalThreshold() public view override returns (uint256) {
        return proposalThreshold;
    }
    
    // quorum: Minimum tokens needed for proposal to pass
    function quorum(
        uint256 blockNumber
    ) public view override returns (uint256) {
        return (token.getPastTotalSupply(blockNumber) * quorum) / 100;
    }
    
    // Execute actions for passed proposal
    function execute(
        address[] calldata targets,
        uint256[] calldata values,
        bytes[] calldata calldatas,
        bytes32 descriptionHash
    ) public payable override returns (uint256) {
        return super.execute(targets, values, calldatas, descriptionHash);
    }
    
    // Proposal state counting
    function proposalState(
        uint256 proposalId
    ) public view override returns (ProposalState) {
        return super.proposalState(proposalId);
    }
    
    // Cast vote with reason
    function castVoteWithReason(
        uint256 proposalId,
        uint8 support,
        string calldata reason
    ) public override returns (uint256) {
        return super.castVoteWithReason(proposalId, support, reason);
    }
}
```

### Hooks

#### Create Proposal

```typescript
// hooks/useCreateProposal.ts
import { useWriteContract, useWaitForTransactionReceipt } from 'wagmi'
import { governorAbi } from '@/abi/governance'

const GOVERNOR_ADDRESS = '0x...'

interface ProposalInput {
  targets: string[]
  values: bigint[]
  signatures: string[]
  calldatas: string[]
  description: string
}

export function useCreateProposal() {
  const { data: hash, writeContract, isPending } = useWriteContract()
  
  const { isLoading: isConfirming, isSuccess } = 
    useWaitForTransactionReceipt({ hash })
  
  function createProposal(input: ProposalInput) {
    writeContract({
      address: GOVERNOR_ADDRESS,
      abi: governorAbi,
      functionName: 'propose',
      args: [
        input.targets,
        input.values,
        input.signatures,
        input.calldatas,
        input.description,
      ],
    })
  }
  
  return { createProposal, hash, isPending, isConfirming, isSuccess }
}
```

#### Cast Vote

```typescript
// hooks/useCastVote.ts
import { useWriteContract, useWaitForTransactionReceipt } from 'wagmi'
import { governorAbi } from '@/abi/governance'

const GOVERNOR_ADDRESS = '0x...'

export enum VoteType {
  Against = 0,
  For = 1,
  Abstain = 2
}

export function useCastVote() {
  const { data: hash, writeContract } = useWriteContract()
  
  const { isLoading: isConfirming } = 
    useWaitForTransactionReceipt({ hash })
  
  function castVote(proposalId: bigint, support: VoteType, reason?: string) {
    if (reason) {
      writeContract({
        address: GOVERNOR_ADDRESS,
        abi: governorAbi,
        functionName: 'castVoteWithReason',
        args: [proposalId, support, reason],
      })
    } else {
      writeContract({
        address: GOVERNOR_ADDRESS,
        abi: governorAbi,
        functionName: 'castVote',
        args: [proposalId, support],
      })
    }
  }
  
  return { castVote, hash, isConfirming }
}
```

### Proposal Card

```typescript
// components/ProposalCard.tsx
'use client'

import { useProposalState } from '@/hooks/useProposalState'
import { useProposalVotes } from '@/hooks/useProposalVotes'
import { useCastVote } from '@/hooks/useCastVote'
import { ProposalState, VoteType } from '@/types/governance'
import { formatUnits } from 'viem'

interface Proposal {
  id: bigint
  description: string
  targets: string[]
}

export function ProposalCard({ proposal }: { proposal: Proposal }) {
  const { state } = useProposalState(proposal.id)
  const { forVotes, againstVotes, abstainVotes } = useProposalVotes(proposal.id)
  const { castVote, isConfirming } = useCastVote()
  
  const totalVotes = forVotes + againstVotes + abstainVotes
  const forPercentage = totalVotes > 0 
    => Number(forVotes) / Number(totalVotes) * 100 
    : 0
  
  const stateLabels: Record<ProposalState, string> = {
    [ProposalState.Pending]: 'Pending',
    [ProposalState.Active]: 'Voting',
    [ProposalState.Canceled]: 'Canceled',
    [ProposalState.Defeated]: 'Defeated',
    [ProposalState.Succeeded]: 'Succeeded',
    [ProposalState.Queued]: 'Queued',
    [ProposalState.Expired]: 'Expired',
    [ProposalState.Executed]: 'Executed',
  }
  
  return (
    <div className="proposal-card">
      <div className="header">
        <h3>Proposal #{proposal.id.toString()}</h3>
        <span className={`status ${stateLabels[state]?.toLowerCase()}`}>
          {stateLabels[state]}
        </span>
      </div>
      
      <p className="description">{proposal.description}</p>
      
      <div className="voting-progress">
        <div className="progress-bar">
          <div 
            className="for" 
            style={{ width: `${forPercentage}%` }}
          />
        </div>
        
        <div className="votes">
          <span>For: {formatUnits(forVotes, 18)}</span>
          <span>Against: {formatUnits(againstVotes, 18)}</span>
          <span>Abstain: {formatUnits(abstainVotes, 18)}</span>
        </div>
      </div>
      
      {state === ProposalState.Active && (
        <div className="vote-buttons">
          <button 
            onClick={() => castVote(proposal.id, VoteType.For)}
            disabled={isConfirming}
          >
            Vote For
          </button>
          <button 
            onClick={() => castVote(proposal.id, VoteType.Against)}
            disabled={isConfirming}
          >
            Vote Against
          </button>
          <button 
            onClick={() => castVote(proposal.id, VoteType.Abstain)}
            disabled={isConfirming}
          >
            Abstain
          </button>
        </div>
      )}
    </div>
  )
}
```

### Create Proposal Form

```typescript
// components/CreateProposalForm.tsx
'use client'

import { useState } from 'react'
import { useCreateProposal } from '@/hooks/useCreateProposal'
import { useAccount } from 'wagmi'

export function CreateProposalForm() {
  const [description, setDescription] = useState('')
  const [target, setTarget] = useState('')
  const [value, setValue] = useState('0')
  const [calldata, setCalldata] = useState('')
  
  const { createProposal, isPending, isConfirming, isSuccess } = useCreateProposal()
  const { address } = useAccount()
  
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    
    createProposal({
      targets: [target],
      values: [BigInt(value)],
      signatures: [''],
      calldatas: [calldata || '0x'],
      description,
    })
  }
  
  if (!address) {
    return <p>Connect wallet to create proposal</p>
  }
  
  return (
    <form onSubmit={handleSubmit} className="proposal-form">
      <h2>Create New Proposal</h2>
      
      <div className="form-group">
        <label>Description</label>
        <textarea
          value={description}
          onChange={(e) => setDescription(e.target.value)}
          placeholder="Describe your proposal..."
          required
        />
      </div>
      
      <div className="form-group">
        <label>Target Address</label>
        <input
          type="text"
          value={target}
          onChange={(e) => setTarget(e.target.value)}
          placeholder="0x..."
          required
        />
      </div>
      
      <div className="form-group">
        <label>Value (ETH)</label>
        <input
          type="text"
          value={value}

      </button>
      
      {isSuccess && <p>Proposal created successfully!</p>}
```

```typescript
    return <div className="animate-pulse">Loading...</div>
  }

  const votingPower = balance ? Number(balance) / Math.pow(10, 18) : 0

  return (
    <div className="flex justify-between items-center">
      <div>
        <p className="text-indigo-200 text-sm">Your Voting Power</p>
        <p className="text-3xl font-bold text-white">
          {votingPower.toLocaleString()} GOV
        </p>
      </div>
      <div className="text-right">
        <p className="text-indigo-200 text-sm">Status</p>
        <p className={`font-semibold ${votingPower > 0 ? 'text-green-400' : 'text-red-400'}`}>
          {votingPower > 0 ? 'Eligible to Vote' : 'Need Tokens'}
        </p>
      </div>
    </div>
  )
}
```
