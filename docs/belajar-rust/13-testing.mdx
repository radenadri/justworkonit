# Testing

Testing adalah bagian integral dari development di Rust. Rust memiliki built-in test framework yang kuat — tidak perlu library eksternal untuk unit testing.

## Menulis Test Pertama

```rust

    a + b
}

fn multiply(a: i32, b: i32) -> i32 {
    a * b
```

## Test Organization

### Unit Tests — di dalam module

```rust
// src/lib.rs atau inline
pub struct Calculator {
    history: Vec<f64>,
}

impl Calculator {
    pub fn new() -> Self {
        Calculator { history: Vec::new() }
    }

    pub fn add(&mut self, a: f64, b: f64) -> f64 {
        let result = a + b;
        self.history.push(result);
        result
    }

    pub fn last_result(&self) -> Option<f64> {
        self.history.last().copied()
    }

    pub fn clear(&mut self) {
        self.history.clear();
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_new_calculator() {
        let calc = Calculator::new();
        assert_eq!(calc.last_result(), None);
    }

    #[test]
    fn test_add() {
        let mut calc = Calculator::new();
        assert_eq!(calc.add(2.0, 3.0), 5.0);
    }

    #[test]
    fn test_history() {
        let mut calc = Calculator::new();
        calc.add(1.0, 2.0);
        calc.add(3.0, 4.0);
        assert_eq!(calc.last_result(), Some(7.0));
    }

    #[test]
    fn test_clear() {
        let mut calc = Calculator::new();
        calc.add(1.0, 1.0);
        calc.clear();
        assert_eq!(calc.last_result(), None);
    }
}

fn main() {}
```

## Studi Kasus: Calculator Library dengan TDD

```rust
#[derive(Debug)]
pub enum CalcError {
    DivisionByZero,
    InvalidExpression(String),
}

impl std::fmt::Display for CalcError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            CalcError::DivisionByZero => write!(f, "Division by zero"),
            CalcError::InvalidExpression(msg) => write!(f, "Invalid: {}", msg),
        }
    }
}

pub struct Calculator {
    memory: f64,
}

impl Calculator {
    pub fn new() -> Self {
        Calculator { memory: 0.0 }
    }

    pub fn add(&self, a: f64, b: f64) -> f64 {
        a + b
    }

    pub fn subtract(&self, a: f64, b: f64) -> f64 {
        a - b
    }

    pub fn multiply(&self, a: f64, b: f64) -> f64 {
        a * b
    }

    pub fn divide(&self, a: f64, b: f64) -> Result<f64, CalcError> {
        if b == 0.0 {
            Err(CalcError::DivisionByZero)
        } else {
            Ok(a / b)
        }
    }

    pub fn store(&mut self, value: f64) {
        self.memory = value;
    }

    pub fn recall(&self) -> f64 {
        self.memory
    }

    pub fn clear_memory(&mut self) {
        self.memory = 0.0;
    }

    pub fn percentage(&self, value: f64, percent: f64) -> f64 {
        value * percent / 100.0
    }

    pub fn evaluate(&self, expr: &str) -> Result<f64, CalcError> {
        let parts: Vec<&str> = expr.split_whitespace().collect();
        if parts.len() != 3 {
            return Err(CalcError::InvalidExpression(
                "Expected format: 'a op b'".to_string(),
            ));
        }

        let a: f64 = parts[0].parse().map_err(|_| {
            CalcError::InvalidExpression(format!("Invalid number: {}", parts[0]))
        })?;

        let b: f64 = parts[2].parse().map_err(|_| {
            CalcError::InvalidExpression(format!("Invalid number: {}", parts[2]))
        })?;

        match parts[1] {
            "+" => Ok(self.add(a, b)),
            "-" => Ok(self.subtract(a, b)),
            "*" => Ok(self.multiply(a, b)),
            "/" => self.divide(a, b),
            op => Err(CalcError::InvalidExpression(format!("Unknown op: {}", op))),
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    fn calc() -> Calculator {
        Calculator::new()
    }

    // === Arithmetic Tests ===

    #[test]
    fn test_add() {
        assert_eq!(calc().add(2.0, 3.0), 5.0);
    }

    #[test]
    fn test_add_negative() {
        assert_eq!(calc().add(-1.0, -2.0), -3.0);
    }

    #[test]
    fn test_subtract() {
        assert_eq!(calc().subtract(10.0, 3.0), 7.0);
    }

    #[test]
    fn test_multiply() {
        assert_eq!(calc().multiply(4.0, 5.0), 20.0);
    }

    #[test]
    fn test_multiply_by_zero() {
        assert_eq!(calc().multiply(100.0, 0.0), 0.0);
    }

    #[test]
    fn test_divide() {
        assert_eq!(calc().divide(10.0, 2.0).unwrap(), 5.0);
    }

    #[test]
    fn test_divide_by_zero() {
        assert!(calc().divide(10.0, 0.0).is_err());
    }

    // === Memory Tests ===

    #[test]
    fn test_memory_default() {
        assert_eq!(calc().recall(), 0.0);
    }

    #[test]
    fn test_memory_store_recall() {
        let mut c = calc();
        c.store(42.0);
        assert_eq!(c.recall(), 42.0);
    }

    #[test]
    fn test_memory_clear() {
        let mut c = calc();
        c.store(42.0);
        c.clear_memory();
        assert_eq!(c.recall(), 0.0);
    }

    // === Percentage Tests ===

    #[test]
    fn test_percentage() {
        assert_eq!(calc().percentage(200.0, 15.0), 30.0);
    }

    // === Expression Tests ===

    #[test]
    fn test_evaluate_add() {
        assert_eq!(calc().evaluate("3 + 4").unwrap(), 7.0);
    }

    #[test]
    fn test_evaluate_divide() {
        assert_eq!(calc().evaluate("10 / 2").unwrap(), 5.0);
    }

    #[test]
    fn test_evaluate_invalid_format() {
        assert!(calc().evaluate("3 +").is_err());
    }

    #[test]
    fn test_evaluate_unknown_operator() {
        assert!(calc().evaluate("3 ^ 4").is_err());
    }

    #[test]
    fn test_evaluate_divide_by_zero() {
        assert!(calc().evaluate("10 / 0").is_err());
    }
}

fn main() {
    let calc = Calculator::new();

    println!("2 + 3 = {}", calc.add(2.0, 3.0));
    println!("10 / 3 = {:.2}", calc.divide(10.0, 3.0).unwrap());
    println!("15% of 200 = {}", calc.percentage(200.0, 15.0));

    match calc.evaluate("42 + 58") {
        Ok(result) => println!("42 + 58 = {}", result),
        Err(e) => println!("Error: {}", e),
    }
}
```

```rust
fn parse_config(line: &str) -> Result<(String, String), String> {
    let parts: Vec<&str> = line.splitn(2, '=').collect();
    if parts.len() != 2 {
```

### Soal 4 — Test Helper Functions (Sedang)

Buat struct `TodoList` dengan add, remove, complete. Tulis tests menggunakan helper function untuk setup.

**Jawaban:**

```rust
#[derive(Debug, Clone)]
struct Todo {
    title: String,
    completed: bool,
}

struct TodoList {
    items: Vec<Todo>,
}

impl TodoList {
    fn new() -> Self {
        TodoList { items: Vec::new() }
    }

    fn add(&mut self, title: &str) {
        self.items.push(Todo {
            title: title.to_string(),
            completed: false,
        });
    }

    fn complete(&mut self, index: usize) -> bool {
        if let Some(todo) = self.items.get_mut(index) {
            todo.completed = true;
            true
        } else {
            false
        }
    }

    fn remove(&mut self, index: usize) -> Option<Todo> {
        if index < self.items.len() {
            Some(self.items.remove(index))
        } else {
            None
        }
    }

    fn pending_count(&self) -> usize {
        self.items.iter().filter(|t| !t.completed).count()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    fn setup() -> TodoList {
        let mut list = TodoList::new();
        list.add("Task 1");
        list.add("Task 2");
        list.add("Task 3");
        list
    }

    #[test]
    fn test_add() {
        let mut list = TodoList::new();
        list.add("Test");
        assert_eq!(list.items.len(), 1);
    }

    #[test]
    fn test_complete() {
        let mut list = setup();
        assert!(list.complete(0));
        assert_eq!(list.pending_count(), 2);
    }

    #[test]
    fn test_complete_invalid() {
        let mut list = setup();
        assert!(!list.complete(99));
    }

    #[test]
    fn test_remove() {
        let mut list = setup();
        let removed = list.remove(1).unwrap();
        assert_eq!(removed.title, "Task 2");
        assert_eq!(list.items.len(), 2);
    }

    #[test]
    fn test_pending_count() {
        let mut list = setup();
        assert_eq!(list.pending_count(), 3);
        list.complete(0);
        list.complete(2);
        assert_eq!(list.pending_count(), 1);
    }
}

fn main() {}
```

```rust
fn validate_password(password: &str) -> Result<(), Vec<String>> {
    let mut errors = Vec::new();

    if password.len() < 8 {
        errors.push("Must be at least 8 characters".to_string());
    }
    if !password.chars().any(|c| c.is_uppercase()) {
        errors.push("Must contain uppercase letter".to_string());
    }
    if !password.chars().any(|c| c.is_numeric()) {
        errors.push("Must contain a number".to_string());
    }

    if errors.is_empty() {
        Ok(())
    } else {
        Err(errors)
    }
}

mod tests {
```

### Soal 8 — Integration Test Setup (Sulit)

Tulis test yang memverifikasi keseluruhan workflow sebuah shopping cart: add items, apply discount, calculate total, checkout.

**Jawaban:**

```rust
struct CartItem {
    name: String,
    price: f64,
    quantity: u32,
}

struct Cart {
    items: Vec<CartItem>,
    discount_percent: f64,
}

impl Cart {
    fn new() -> Self {
        Cart { items: Vec::new(), discount_percent: 0.0 }
    }

    fn add_item(&mut self, name: &str, price: f64, qty: u32) {
        self.items.push(CartItem {
            name: name.to_string(),
            price,
            quantity: qty,
        });
    }

    fn apply_discount(&mut self, percent: f64) {
        self.discount_percent = percent;
    }

    fn subtotal(&self) -> f64 {
        self.items.iter().map(|i| i.price * i.quantity as f64).sum()
    }

    fn total(&self) -> f64 {
        let sub = self.subtotal();
        sub - (sub * self.discount_percent / 100.0)
    }

    fn checkout(&mut self) -> f64 {
        let total = self.total();
        self.items.clear();
        self.discount_percent = 0.0;
        total
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_full_workflow() {
        let mut cart = Cart::new();

        // Add items
        cart.add_item("Laptop", 1000.0, 1);
        cart.add_item("Mouse", 25.0, 2);
        assert_eq!(cart.items.len(), 2);

        // Check subtotal
        assert_eq!(cart.subtotal(), 1050.0);

        // Apply discount
        cart.apply_discount(10.0);
        assert_eq!(cart.total(), 945.0);

        // Checkout
        let paid = cart.checkout();
        assert_eq!(paid, 945.0);
        assert!(cart.items.is_empty());
        assert_eq!(cart.discount_percent, 0.0);
    }

    #[test]
    fn test_empty_cart() {
        let cart = Cart::new();
        assert_eq!(cart.total(), 0.0);
    }
}

fn main() {}
```
