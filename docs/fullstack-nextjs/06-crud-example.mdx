## 📝 Zod Validation Schemas

### src/lib/validations/post.ts

```typescript
import { z } from 'zod';

// Schema untuk validasi input
export const createPostSchema = z.object({
  title: z.string()
    .min(3, 'Title minimal 3 karakter')
    .max(100, 'Title maksimal 100 karakter'),
  content: z.string()
    .min(10, 'Content minimal 10 karakter'),
  published: z.boolean().default(false),
});

// Schema untuk update (semua fields optional)
export const updatePostSchema = createPostSchema.partial();

// Schema untuk parameter ID
export const postIdSchema = z.object({
  id: z.coerce.number().int().positive('ID harus angka positif'),
});

// Type inferences
export type CreatePostInput = z.infer<typeof createPostSchema>;
export type UpdatePostInput = z.infer<typeof updatePostSchema>;
export type PostIdInput = z.infer<typeof postIdSchema>;
```

## 🎯 Server Actions

### src/actions/posts.ts

```typescript
"use server";

import { db } from "@/lib/db";
import { posts } from "@/lib/db/schema";
import { eq, desc } from "drizzle-orm";
import { revalidatePath } from "next/cache";
import { 
  createPostSchema, 
  updatePostSchema, 
  postIdSchema 
} from "@/lib/validations/post";

export async function getPosts() {
  const allPosts = await db
    .select()
    .from(posts)
    .orderBy(desc(posts.createdAt));
  
  return allPosts;
}

export async function getPost(id: number) {
  const post = await db
    .select()
    .from(posts)
    .where(eq(posts.id, id))
    .limit(1);
  
  return post[0] || null;
}

export async function createPost(formData: FormData) {
  const rawData = {
    title: formData.get("title"),
    content: formData.get("content"),
    published: formData.get("published") === "on",
  };
  
  // Validasi dengan Zod
  const result = createPostSchema.safeParse(rawData);
  
  if (!result.success) {
    return {
      success: false,
      errors: result.error.flatten().fieldErrors,
    };
  }
  
  const { title, content, published } = result.data;
  
  const newPost = await db
    .insert(posts)
    .values({
      title,
      content,
      published,
    })
    .returning();
  
  revalidatePath("/posts");
  
  return { success: true, data: newPost[0] };
}

export async function updatePost(formData: FormData) {
  const rawData = {
    id: formData.get("id"),
    title: formData.get("title"),
    content: formData.get("content"),
    published: formData.get("published") === "on",
  };
  
  // Validasi dengan Zod
  const idResult = postIdSchema.safeParse({ id: rawData.id });
  if (!idResult.success) {
    return { success: false, errors: idResult.error.flatten().fieldErrors };
  }
  
  const dataResult = updatePostSchema.safeParse(rawData);
  if (!dataResult.success) {
    return { success: false, errors: dataResult.error.flatten().fieldErrors };
  }
  
  const { id } = idResult.data;
  const { title, content, published } = dataResult.data;
  
  const updatedPost = await db
    .update(posts)
    .set({
      ...(title && { title }),
      ...(content && { content }),
      ...(published !== undefined && { published }),
      updatedAt: new Date(),
    })
    .where(eq(posts.id, id))
    .returning();
  
  revalidatePath("/posts");
  
  return { success: true, data: updatedPost[0] };
}

export async function deletePost(id: number) {
  // Validasi ID
  const idResult = postIdSchema.safeParse({ id });
  
  if (!idResult.success) {
    return { success: false, errors: idResult.error.flatten().fieldErrors };
  }
  
  await db
    .delete(posts)
    .where(eq(posts.id, idResult.data.id));
  
  revalidatePath("/posts");
  
  return { success: true };
}

export async function togglePublish(formData: FormData) {
  const rawData = {
    id: formData.get("id"),
    published: formData.get("published") === "true",
  };
  
  const idResult = postIdSchema.safeParse({ id: rawData.id });
  if (!idResult.success) {
    return { success: false, errors: idResult.error.flatten().fieldErrors };
  }
  
  const { id } = idResult.data;
  const published = rawData.published;
  
  await db
    .update(posts)
    .set({ published })
    .where(eq(posts.id, id));
  
  revalidatePath("/posts");
  
  return { success: true };
}
```

## 🗃️ Zustand Store

### src/stores/use-post-store.ts

```typescript
import { create } from 'zustand';

interface Post {
  id: number;
  title: string;
  content: string;
  published: boolean;
  createdAt: Date;
  updatedAt: Date;
}

interface PostState {
  posts: Post[];
  isLoading: boolean;
  filter: 'all' | 'published' | 'draft';
  setPosts: (posts: Post[]) => void;
  addPost: (post: Post) => void;
  updatePost: (id: number, updates: Partial<Post>) => void;
  removePost: (id: number) => void;
  setFilter: (filter: 'all' | 'published' | 'draft') => void;
  setLoading: (loading: boolean) => void;
}

export const usePostStore = create<PostState>((set) => ({
  posts: [],
  isLoading: false,
  filter: 'all',
  
  setPosts: (posts) => set({ posts }),
  
  addPost: (post) => set((state) => ({ 
    posts: [post, ...state.posts] 
  })),
  
  updatePost: (id, updates) => set((state) => ({
    posts: state.posts.map((post) =>
      post.id === id ? { ...post, ...updates } : post
    ),
  })),
  
  removePost: (id) => set((state) => ({
    posts: state.posts.filter((post) => post.id !== id),
  })),
  
  setFilter: (filter) => set({ filter }),
  
  setLoading: (isLoading) => set({ isLoading }),
}));
```

## 🎨 UI Components

### src/components/posts/post-form.tsx (dengan Zod + React Hook Form)

```typescript
"use client";

import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Textarea } from "@/components/ui/textarea";
import { usePostStore } from "@/stores/use-post-store";
import { createPostSchema } from "@/lib/validations/post";

type FormData = z.infer<typeof createPostSchema>;

export function PostForm() {
  const addPost = usePostStore((state) => state.addPost);
  
  const {
    register,
    handleSubmit,
    reset,
    formState: { errors, isSubmitting },
  } = useForm<FormData>({
    resolver: zodResolver(createPostSchema),
    defaultValues: {
      title: '',
      content: '',
      published: false,
    },
  });
  
  const onSubmit = async (data: FormData) => {
    try {
      const formData = new FormData();
      formData.append("title", data.title);
      formData.append("content", data.content);
      formData.append("published", String(data.published));
      
      const { createPost } = await import("@/actions/posts");
      const result = await createPost(formData);
      
      if (result.success && result.data) {
        addPost(result.data);
        reset();
      }
    } catch (error) {
      console.error("Failed to create post", error);
    }
  };
  
  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
      <div>
        <Input
          {...register('title')}
          placeholder="Post title"
        />
        {errors.title && (
          <p className="text-sm text-red-500 mt-1">
            {errors.title.message}
          </p>
        )}
      </div>
      <div>
        <Textarea
          {...register('content')}
          placeholder="Post content"
        />
        {errors.content && (
          <p className="text-sm text-red-500 mt-1">
            {errors.content.message}
          </p>
        )}
      </div>
      <div className="flex items-center gap-2">
        <input
          type="checkbox"
          id="published"
          {...register('published')}
          className="w-4 h-4"
        />
        <label htmlFor="published" className="text-sm">
          Publish immediately
        </label>
      </div>
      <Button type="submit" disabled={isSubmitting}>
        {isSubmitting ? "Creating..." : "Create Post"}
      </Button>
    </form>
  );
}
```

### src/components/posts/post-item.tsx

```typescript
"use client";

import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Checkbox } from "@/components/ui/checkbox";
import { usePostStore } from "@/stores/use-post-store";
import { formatDistanceToNow } from "date-fns";

interface PostItemProps {
  post: {
    id: number;
    title: string;
    content: string;
    published: boolean;
    createdAt: Date;
  };
}

export function PostItem({ post }: PostItemProps) {
  const removePost = usePostStore((state) => state.removePost);
  const updatePost = usePostStore((state) => state.updatePost);
  
  const handleToggle = async () => {
    try {
      const { togglePublish } = await import("@/actions/posts");
      await togglePublish(post.id, !post.published);
      updatePost(post.id, { published: !post.published });
    } catch (error) {
      console.error("Failed to toggle", error);
    }
  };
  
  const handleDelete = async () => {
    if (!confirm("Are you sure?")) return;
    
    try {
      const { deletePost } = await import("@/actions/posts");
      await deletePost(post.id);
      removePost(post.id);
    } catch (error) {
      console.error("Failed to delete", error);
    }
  };
  
  return (
    <Card className={post.published ? "border-green-500" : ""}>
      <CardHeader className="pb-2">
        <div className="flex items-start justify-between">
          <CardTitle className="text-lg">{post.title}</CardTitle>
          <Checkbox
            checked={post.published}
            onCheckedChange={handleToggle}
          />
        </div>
      </CardHeader>
      <CardContent>
        <p className="text-sm text-gray-600 mb-4">{post.content}</p>
        <div className="flex items-center justify-between">
          <span className="text-xs text-gray-400">
            {formatDistanceToNow(new Date(post.createdAt), { addSuffix: true })}
          </span>
          <Button
            variant="destructive"
            size="sm"
            onClick={handleDelete}
          >
            Delete
          </Button>
        </div>
      </CardContent>
    </Card>
  );
}
```

### src/components/posts/post-list.tsx

```typescript
"use client";

import { usePostStore } from "@/stores/use-post-store";
import { PostItem } from "./post-item";
import { useEffect, useState } from "react";

export function PostList() {
  const { posts, filter, setPosts, isLoading, setLoading } = usePostStore();
  const [mounted, setMounted] = useState(false);
  
  useEffect(() => {
    setMounted(true);
    
    async function fetchPosts() {
      setLoading(true);
      try {
        const { getPosts } = await import("@/actions/posts");
        const data = await getPosts();
        setPosts(data);
      } catch (error) {
        console.error("Failed to fetch posts", error);
      } finally {
        setLoading(false);
      }
    }
    
    fetchPosts();
  }, [setPosts, setLoading]);
  
  const filteredPosts = posts.filter((post) => {
    if (filter === 'published') return post.published;
    if (filter === 'draft') return !post.published;
    return true;
  });
  
  if (!mounted || isLoading) {
    return <div>Loading...</div>;
  }
  
  if (filteredPosts.length === 0) {
    return (
      <div className="text-center py-8 text-gray-500">
        No posts yet. Create your first post!
      </div>
    );
  }
  
  return (
    <div className="space-y-4">
      {filteredPosts.map((post) => (
        <PostItem key={post.id} post={post} />
      ))}
    </div>
  );
}
```
