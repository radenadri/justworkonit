
### Validasi String

```typescript
const stringValidations = z.string()
  .min(3, 'Minimal 3 karakter')
  .max(20, 'Maksimal 20 karakter')
  .length(10, 'Harus tepat 10 karakter')
  .email('Email tidak valid')
  .url('URL tidak valid')
  .uuid('UUID tidak valid')
  .regex(/^[a-zA-Z0-9]+$/, 'Hanya alphanumeric')
  .startsWith('Hello')
  .endsWith('World')
  .trim() // menghapus whitespace
  .toLowerCase()
  .toUpperCase();
```

### Validasi Number

```typescript
const numberValidations = z.number()
  .min(0, 'Tidak boleh negatif')

  .positive('Harus positif')
  .negative('Harus negatif')
  .nonnegative('Tidak boleh negatif')

  .multipleOf(5, 'Harus kelipatan 5');

// Optional: custom error message
const constrainedNumber = z.number({ 
  invalid_type_error: 'Harus berupa angka',
  required_error: 'Wajib diisi'
});
```

### Validasi Boolean

```typescript

```

## ✨ Custom Validation

### Basic Refinement

```typescript
const usernameSchema = z.string()
  .min(3)
  .max(20)
  .refine(
    (val) => /^[a-zA-Z0-9_]+$/.test(val),
    { message: 'Hanya alphanumeric dan underscore' }
  );
```

### SuperRefine untuk Complex Logic

```typescript
const passwordSchema = z.string().superRefine((val, ctx) => {
  if (val.length < 8) {
    ctx.addIssue({
      code: z.ZodIssueCode.too_small,
      minimum: 8,
      type: 'string',
      message: 'Password minimal 8 karakter',
    });
  }
  
  if (!/[A-Z]/.test(val)) {
    ctx.addIssue({
      code: z.ZodIssueCode.custom,
      message: 'Harus ada huruf besar',
    });
  }
  
  if (!/[a-z]/.test(val)) {
    ctx.addIssue({
      code: z.ZodIssueCode.custom,
      message: 'Harus ada huruf kecil',
    });
  }
  
  if (!/[0-9]/.test(val)) {
    ctx.addIssue({
      code: z.ZodIssueCode.custom,
      message: 'Harus ada angka',
    });
  }
});
```

### Async Validation

```typescript
const asyncSchema = z.string().refine(
  async (val) => {
    // Simulate API call
    const response = await fetch(`/api/check-username?username=${val}`);
    const data = await response.json();
    return data.available;
  },
  { message: 'Username sudah digunakan' }
);
```

## 🎯 Type Inference

### Extract Types dari Schema

```typescript
const userSchema = z.object({

  name: z.string(),
  email: z.string().email(),
  age: z.number().optional(),
  createdAt: z.date(),

// Infer TypeScript type

//   id: number;

```

### Input vs Output Types

```typescript
const transformSchema = z.string().transform((val) => val.length);

type Input = z.input<typeof transformSchema>;  // string
type Output = z.output<typeof transformSchema>; // number
```

## ⚠️ Error Handling

### Safe Parse

```typescript
const userSchema = z.object({
  name: z.string().min(3),
  email: z.string().email(),
});

const result = userSchema.safeParse({ 
  name: 'Jo', 
  email: 'invalid' 
});

if (!result.success) {
  console.log(result.error.format());
  // {
  //   name: ['String must contain at least 3 character(s)'],
  //   email: ['Invalid email']
  // }
}
```

### Custom Error Messages

```typescript
const schema = z.object({
  name: z.string({ 
    required_error: 'Nama wajib diisi',
    invalid_type_error: 'Nama harus berupa string'
  }),
  age: z.number({ 
    invalid_type_error: 'Umur harus berupa angka' 
  }).min(18, { 
    message: 'Minimal 18 tahun' 
  }),
});
```

### Pretty Print Errors (Zod 4)

```typescript
const result = schema.safeParse(data);

if (!result.success) {
  console.log(result.error.pretty());
  // Atau
  console.log(result.error.flatten());
}
```

## 📝 Zod dengan React Hook Form

### Basic Form

```typescript
"use client";

import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';

const formSchema = z.object({
  name: z.string().min(2, 'Nama minimal 2 karakter'),
  email: z.string().email('Email tidak valid'),
  password: z.string()
    .min(8, 'Password minimal 8 karakter')
    .regex(/[A-Z]/, 'Harus ada huruf besar')
    .regex(/[a-z]/, 'Harus ada huruf kecil')
    .regex(/[0-9]/, 'Harus ada angka'),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  message: 'Password tidak cocok',
  path: ['confirmPassword'],
});

type FormData = z.infer<typeof formSchema>;

export function RegisterForm() {
  const form = useForm<FormData>({
    resolver: zodResolver(formSchema),
    defaultValues: {
      name: '',
      email: '',
      password: '',
      confirmPassword: '',
    },
  });

  function onSubmit(data: FormData) {
    console.log(data);
  }

  return (
    <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
      <div>
        <Input {...form.register('name')} placeholder="Nama" />
        {form.formState.errors.name && (
          <p className="text-red-500 text-sm">
            {form.formState.errors.name.message}
          </p>
        )}
      </div>
      
      <div>
        <Input {...form.register('email')} placeholder="Email" />
        {form.formState.errors.email && (
          <p className="text-red-500 text-sm">
            {form.formState.errors.email.message}
          </p>
        )}
      </div>
      
      <div>
        <Input 
          {...form.register('password')} 
          type="password" 
          placeholder="Password" 
        />
        {form.formState.errors.password && (
          <p className="text-red-500 text-sm">
            {form.formState.errors.password.message}
          </p>
        )}
      </div>
      
      <div>
        <Input 
          {...form.register('confirmPassword')} 
          type="password" 
          placeholder="Konfirmasi Password" 
        />
        {form.formState.errors.confirmPassword && (
          <p className="text-red-500 text-sm">
            {form.formState.errors.confirmPassword.message}
          </p>
        )}
      </div>
      
      <Button type="submit">Register</Button>
    </form>
  );
}
```

## 🔒 Server Actions Validation

### Validasi di Server Action

```typescript
"use server";

import { z } from 'zod';

const createPostSchema = z.object({
  title: z.string()
    .min(3, 'Title minimal 3 karakter')
    .max(100, 'Title maksimal 100 karakter'),
  content: z.string()
    .min(10, 'Content minimal 10 karakter'),
  published: z.boolean().default(false),
  tags: z.array(z.string()).max(5, 'Max 5 tags'),
});

export async function createPost(formData: FormData) {
  const rawData = {
    title: formData.get('title'),
    content: formData.get('content'),
    published: formData.get('published') === 'on',
    tags: formData.get('tags')?.toString().split(',') || [],
  };
  
  const result = createPostSchema.safeParse(rawData);
  
  if (!result.success) {
    return { 
      success: false, 
      errors: result.error.flatten().fieldErrors 
    };
  }
  
  // Validasi berhasil, lanjutkan dengan logic
  const { title, content, published, tags } = result.data;
  
  // ... Simpan ke database
  return { success: true };
}
```

## 🎯 Zod 4 New Features Summary

| Feature | Description |
|---------|-------------|
| **@zod/core** | Modular architecture |
| **.overwrite()** | For partial updates |
| **Refinements Inside Schemas** | Cleaner validation logic |

| **stringbool** | Environment variable parsing |

| **Template Literal Types** | Pattern-based strings |
| **Custom Email Regex** | Customizable email validation |
| **Error pretty-printing** | Better error formatting |
| **i18n Support** | Internationalization |
| **File Schemas** | File validation |
| **z.interface()** | Alternative to z.object() |
| **JSON Schema Conversion** | Export to JSON Schema |
| **Global Registry** | Reusable schemas |
